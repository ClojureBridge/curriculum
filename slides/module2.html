Vector functions

(vector? [5 10 15])
;;=> true

(vector 5 10 15)
;;=> [5 10 15]

(conj [5 10] 15)
;;=> [5 10 15]

Vector functions

(count [5 10 15])
;;=> 3

(nth [5 10 15] 1)
;;=> 10

(first [5 10 15])
;;=> 5
